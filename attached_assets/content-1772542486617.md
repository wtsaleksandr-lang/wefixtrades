type ImpactLevel = "high" | "medium" | "low";
type MetricStatus = "good" | "needs" | "critical";

function impactToLevel(impact: any): ImpactLevel {
  const s = String(impact ?? "").toLowerCase();
  if (s.includes("high") || s.includes("critical")) return "high";
  if (s.includes("med") || s.includes("warn")) return "medium";
  if (s.includes("low") || s.includes("minor")) return "low";
  const n = typeof impact === "number" ? impact : Number(impact);
  if (!Number.isNaN(n)) {
    if (n >= 8) return "high";
    if (n >= 5) return "medium";
    return "low";
  }
  return "medium";
}

function fmtMs(v: any): string {
  const n = Number(v);
  if (Number.isNaN(n)) return String(v ?? "");
  if (n >= 1000) return `${(n / 1000).toFixed(2)}s`;
  return `${Math.round(n)}ms`;
}

function fmtSec(v: any): string {
  const n = Number(v);
  if (Number.isNaN(n)) return String(v ?? "");
  if (n < 1) return `${Math.round(n * 1000)}ms`;
  return `${n.toFixed(2)}s`;
}

function metricStatus(name: "FCP" | "LCP" | "TBT" | "CLS", raw: any): MetricStatus {
  const v = Number(raw);
  if (Number.isNaN(v)) return "needs";
  if (name === "FCP") {
    if (v <= 1.8) return "good";
    if (v <= 3.0) return "needs";
    return "critical";
  }
  if (name === "LCP") {
    if (v <= 2.5) return "good";
    if (v <= 4.0) return "needs";
    return "critical";
  }
  if (name === "TBT") {
    if (v <= 200) return "good";
    if (v <= 600) return "needs";
    return "critical";
  }
  // CLS
  if (v <= 0.1) return "good";
  if (v <= 0.25) return "needs";
  return "critical";
}

function statusLabel(s: MetricStatus): string {
  if (s === "good") return "Good";
  if (s === "needs") return "Needs work";
  return "Critical";
}

type ImpactLevel = "high" | "medium" | "low";
type MetricStatus = "good" | "needs" | "critical";

function impactToLevel(impact: any): ImpactLevel {
  const s = String(impact ?? "").toLowerCase();
  if (s.includes("high") || s.includes("critical")) return "high";
  if (s.includes("med") || s.includes("warn")) return "medium";
  if (s.includes("low") || s.includes("minor")) return "low";
  const n = typeof impact === "number" ? impact : Number(impact);
  if (!Number.isNaN(n)) {
    if (n >= 8) return "high";
    if (n >= 5) return "medium";
    return "low";
  }
  return "medium";
}

function fmtMs(v: any): string {
  const n = Number(v);
  if (Number.isNaN(n)) return String(v ?? "");
  if (n >= 1000) return `${(n / 1000).toFixed(2)}s`;
  return `${Math.round(n)}ms`;
}

function fmtSec(v: any): string {
  const n = Number(v);
  if (Number.isNaN(n)) return String(v ?? "");
  if (n < 1) return `${Math.round(n * 1000)}ms`;
  return `${n.toFixed(2)}s`;
}

function metricStatus(name: "FCP" | "LCP" | "TBT" | "CLS", raw: any): MetricStatus {
  const v = Number(raw);
  if (Number.isNaN(v)) return "needs";
  if (name === "FCP") {
    if (v <= 1.8) return "good";
    if (v <= 3.0) return "needs";
    return "critical";
  }
  if (name === "LCP") {
    if (v <= 2.5) return "good";
    if (v <= 4.0) return "needs";
    return "critical";
  }
  if (name === "TBT") {
    if (v <= 200) return "good";
    if (v <= 600) return "needs";
    return "critical";
  }
  // CLS
  if (v <= 0.1) return "good";
  if (v <= 0.25) return "needs";
  return "critical";
}

function statusLabel(s: MetricStatus): string {
  if (s === "good") return "Good";

  return "Critical";
}

                <div className={`${reportStyles.body} ${reportStyles.muted}`}>Your Google Maps Profile</div>
              </div>
              <span className={reportStyles.pill}>{business?.status ?? "N/A"}</span>

                  <span className={reportStyles.kvIcon}><Icon kind="star" /></span>
                  <span><b>{rating ?? "--"}</b></span>
                  <span className={reportStyles.muted}>{reviews != null ? `${reviews} reviews` : ""}</span>
                </div>
              )}
              {address && (
                <div className={reportStyles.kvRow}>
                  <span className={reportStyles.kvIcon}><Icon kind="pin" /></span>
                  <span className={reportStyles.body}>{address}</span>
                </div>

                <div className={reportStyles.kvRow}>
                  <span className={reportStyles.kvIcon}><Icon kind="phone" /></span>
                  <span className={reportStyles.body}>{phone}</span>
                </div>

                <div className={reportStyles.kvRow}>
                  <span className={reportStyles.kvIcon}><Icon kind="globe" /></span>
                  <a className={reportStyles.body} href={website} target="_blank" rel="noreferrer">{String(website).replace(/^https?:\/\//, "")}</a>

              )}

                <div className={reportStyles.kvRow}>
                  <span className={reportStyles.kvIcon}><Icon kind="clock" /></span>
                  <span className={reportStyles.body}>Opening Hours</span>
                </div>
              )}
            </div>

            {photos.length > 0 && (
              <div className={reportStyles.photos}>
                {photos.slice(0, 4).map((src, i) => (
                  <img key={i} className={reportStyles.photo} src={src} alt={`photo-${i}`} />
                ))}
              </div>
            )}
          </div>

          {/* TABS */}
          <div className={reportStyles.tabsWrap}>
            <button
              type="button"
              className={`${reportStyles.tabBtn} ${activeTab === "maps" ? reportStyles.tabBtnActive : ""}`}
              onClick={() => setActiveTab("maps")}
            >
              Google Maps Audit
            </button>
            <button
              type="button"
              className={`${reportStyles.tabBtn} ${activeTab === "speed" ? reportStyles.tabBtnActive : ""}`}
              onClick={() => setActiveTab("speed")}
            >
              Website SEO &amp; Speed
            </button>
          </div>

          {/* MAPS TAB */}
          {activeTab === "maps" && (
            <div className={reportStyles.stack}>
              <div className={reportStyles.sectionTitle}>Visibility Opportunities Detected</div>

              <div className={reportStyles.issueList}>
                {(mapsIssues ?? []).map((it, idx) => {
                  const level = impactToLevel(it?.impact ?? it?.severity ?? it?.level);
                  const cls =
                    level === "high"
                      ? `${reportStyles.issueCard} ${reportStyles.high}`
                      : level === "medium"
                      ? `${reportStyles.issueCard} ${reportStyles.med}`
                      : `${reportStyles.issueCard} ${reportStyles.low}`;

                  const pillText = level === "high" ? "High Impact" : level === "medium" ? "Medium Impact" : "Low Impact";

                  return (
                    <div key={idx} className={cls}>
                      <div className={reportStyles.issueIconBox}><Icon kind="issue" /></div>
                      <div className={reportStyles.issueMain}>
                        <div className={reportStyles.issueTitle}>{it?.title ?? it?.name ?? "Opportunity"}</div>
                        <div className={reportStyles.issueDesc}>{it?.description ?? it?.desc ?? it?.message ?? ""}</div>
                      </div>
                      <div className={reportStyles.impactPill}>{pillText}</div>
                    </div>
                  );
                })}
              </div>
            </div>
          )}

          {/* SPEED TAB */}
          {activeTab === "speed" && (
            <div className={reportStyles.stack}>
              <div className={reportStyles.infoBox}>
                <div className={reportStyles.infoIcon}><Icon kind="info" /></div>
                <div>
                  <div className={reportStyles.h4}>Speed Impact</div>
                  <div className={`${reportStyles.body} ${reportStyles.muted}`}>Your website speed may be limiting conversions and local rankings.</div>
                </div>
              </div>

              <PerfBlock
                label="Mobile"
                kind="mobile"
                tone="red"
                score={mobile?.score ?? mobile?.performanceScore ?? mobile?.performance ?? mobile?.perfScore}
                data={mobile}
              />

              <PerfBlock
                label="Desktop"
                kind="desktop"
                tone="amber"
                score={desktop?.score ?? desktop?.performanceScore ?? desktop?.performance ?? desktop?.perfScore}
                data={desktop}
              />
            </div>
          )}

          {/* CTA */}
          <div className={`${reportStyles.card} ${reportStyles.cardPad} ${reportStyles.ctaCard}`}>
            <div className={reportStyles.ctaTitle}>Ready to fix your local visibility?</div>
            <div className={reportStyles.ctaSub}>Professional optimization for both Google Maps and your website.</div>
            <div className={reportStyles.ctaBtns}>
              <button type="button" className={reportStyles.btnPrimary} onClick={onFixClick}>
                Fix This For Me &nbsp; →
              </button>
              <button type="button" className={reportStyles.btnSecondary} onClick={onEmailClick}>
                Email Me This Report
              </button>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}

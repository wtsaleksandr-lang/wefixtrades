PART 2 — ISSUES TO FIX (per user description)
====================================================================

1. COLOR MISMATCH
   The cyan color (#00D4C8) must match the headline "On autopilot" color.
   Check what color that headline uses — should be same cyan.
   Both the cyan squares AND the hovered button must use identical color.

2. ARROWS MISSING / NOT VISIBLE
   The arrow icon in the cyan square is not rendering or invisible.
   The arrow must be dark (#0d1514 or near-black) on the cyan background.
   NOT white. Dark arrow on cyan = visible. White arrow on cyan = invisible.

3. CYAN SQUARE: SHOULD HAVE A GAP FROM BUTTON EDGE
   The square should NOT touch the outer border of the dark pill.
   There must be ~4px of padding/gap between the square and the button's edges.
   Think of it as an inset square with the dark button background visible
   as a thin border all around the square.
   
   Implementation: the cyan square sits INSIDE a padding area.
   OR: the square has `margin: 4px` inside its container.

4. NO TRANSPARENT EFFECT ON BUTTONS
   Remove any opacity, backdrop-filter, or rgba transparency from the button.
   The dark button background must be a solid color: #1a1f1e or #1c2220
   NOT rgba(255,255,255,0.06) or any semi-transparent value.

5. FONT STYLE MUST MATCH EFFORTEL
   Effortel nav buttons use:
     font-size: 11px
     font-weight: 700
     letter-spacing: 0.12em
     text-transform: uppercase
     font-family: same as site heading font (check what's loaded)
   
   The text in Image 1 appears to match — verify it's using the correct
   font and not falling back to a system font.

6. HOVER ANIMATION IS WRONG — currently expands full width
   The button should NOT expand to fill its container/parent width.
   It should expand ONLY to a fixed target width that fits text + arrow.
   
   Target width on hover = button text width + arrow area (~44px) + padding
   
   Do NOT use width: 100% on hover.
   Use a calculated fixed width or use fit-content with a max.

7. PRODUCT = CYAN, SERVICES = DARK (default states differ)
   Looking again at Image 1 more carefully:
   Both buttons appear similar in normal state.
   But the PRODUCT button may have slight cyan tint as default.
   
   ACTUALLY from Image 3: PRODUCT ends up cyan after hover.
   The normal state shows PRODUCT as dark, same as SERVICES.
   The difference is which one you hover.

const { chromium } = require('/home/claude/.npm-global/lib/node_modules/playwright');
const fs = require('fs');

(async () => {
  const browser = await chromium.launch({ headless: true });
  const page = await browser.newPage();
  await page.setViewportSize({ width: 1440, height: 900 });
  
  await page.goto('http://localhost:3000', { waitUntil: 'networkidle' });
  await page.waitForTimeout(2000);

  fs.mkdirSync('./btn-verify', { recursive: true });

  // Full page normal
  await page.screenshot({ path: './btn-verify/01-normal-full.png' });

  // Find buttons
  const buttons = await page.$$('.effortel-btn, [class*="nav-btn"], [class*="NavButton"]');
  console.log(`Found ${buttons.length} buttons`);

  for (let i = 0; i < buttons.length; i++) {
    const btn = buttons[i];
    const visible = await btn.isVisible();
    if (!visible) continue;

    const box = await btn.boundingBox();
    const CROP_PADDING = 20;
    const clip = {
      x: Math.max(0, box.x - CROP_PADDING),
      y: Math.max(0, box.y - CROP_PADDING),
      width: box.width + CROP_PADDING * 4,
      height: box.height + CROP_PADDING * 2
    };

    // --- Normal state ---
    await page.screenshot({ path: `./btn-verify/btn${i}-1-normal.png`, clip });

    // --- Get computed styles normal ---
    const normalStyles = await page.evaluate((el) => {
      const s = window.getComputedStyle(el);
      const sq = el.querySelector('[class*="arrow"], [class*="square"]');
      const sqS = sq ? window.getComputedStyle(sq) : null;
      return {
        btnBg: s.backgroundColor,
        btnHeight: s.height,
        btnBorderRadius: s.borderRadius,
        btnOpacity: s.opacity,
        btnBorder: s.border,
        squareBg: sqS?.backgroundColor,
        squareWidth: sqS?.width,
        squareHeight: sqS?.height,
        squareBorderRadius: sqS?.borderRadius,
        text: el.textContent?.trim().substring(0, 30),
      };
    }, btn);
    console.log(`\nButton ${i}: "${normalStyles.text}"`);
    console.log('  Normal state:');
    console.log('  BG:', normalStyles.btnBg, normalStyles.btnBg.includes('0, 0, 0, 0') ? '⚠️ TRANSPARENT!' : '✅');
    console.log('  Height:', normalStyles.btnHeight, normalStyles.btnHeight === '46px' ? '✅' : '⚠️ expected 46px');
    console.log('  Border-radius:', normalStyles.btnBorderRadius);
    console.log('  Arrow square:', normalStyles.squareBg, normalStyles.squareWidth, '×', normalStyles.squareHeight);

    // --- Hover mid (capture at 150ms) ---
    await btn.hover();
    await page.waitForTimeout(150);
    await page.screenshot({ path: `./btn-verify/btn${i}-2-hover-mid.png`, clip });

    // --- Hover complete ---
    await page.waitForTimeout(300);
    await page.screenshot({ path: `./btn-verify/btn${i}-3-hover-done.png`, clip });

    // --- Get hover styles ---
    const hoverStyles = await page.evaluate((el) => {
      const s = window.getComputedStyle(el);
      return {
        btnBg: s.backgroundColor,
        textColor: window.getComputedStyle(el.querySelector('[class*="text"], span'))?.color,
      };
    }, btn);
    console.log('  Hover BG:', hoverStyles.btnBg, hoverStyles.btnBg.includes('0, 212, 200') ? '✅ cyan' : '⚠️ not cyan');

    // --- Reset ---
    await page.mouse.move(100, 500);
    await page.waitForTimeout(400);
  }

  // Verify cyan color matches headline
  const headlineColor = await page.evaluate(() => {
    const h1 = document.querySelector('h1, h2, [class*="headline"]');
    if (!h1) return null;
    // Find cyan-colored text (span, em, etc.)
    const spans = h1.querySelectorAll('span, em, strong');
    for (const span of spans) {
      const color = window.getComputedStyle(span).color;
      if (color.includes('0, 212') || color.includes('0, 180') || color.includes('turquoise')) {
        return { element: span.tagName, color };
      }
    }
    return { headlineFull: window.getComputedStyle(h1).color };
  });
  console.log('\nHeadline cyan color:', JSON.stringify(headlineColor));
  console.log('Button cyan should match: #00D4C8 = rgb(0, 212, 200)');

  await browser.close();
  console.log('\nDone! Check ./btn-verify/ for screenshots.');
  console.log('Compare btn0-1-normal, btn0-2-hover-mid, btn0-3-hover-done');
})();
